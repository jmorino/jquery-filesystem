 83 		/**
 84 		 * Requests a filesystem in which to store application data.
 85 		 * 
 86 		 * @function
 87 		 * @name jQuery.filesystem#requestFS
 88 		 * @return {jQuery.Deferred} a request object containing the result {@link jQuery.filesystem.FileSystem} object in case of success
 89 		 */
 90 		filesystem.prototype.requestFS = function(eType, iSize) {
 91 			
 92 			var oDef = $.Deferred();
 93 			window.requestFileSystem(eType, iSize,
 94 				$.proxy(function(oD, oFileSystem) { oD.resolve(new filesystem.FileSystem(oFileSystem)); }, null, oDef),
 95 				filesystem._fnErrorCallback(oDef));
 96 			
 97 			return oDef;
 98 		};

457 		/**
458 		 * Lists files and directories in a directory.
459 		 * 
460 		 * @function
461 		 * @name jQuery.filesystem.DirectoryEntry#listContent
462 		 * @return {jQuery.Deferred} a request object containing the result {@link jQuery.filesystem.Entry[]} array in case of success
463 		 */
464 		IDEntry.prototype.listContent = function() {
465 			
466 			var oDef = $.Deferred(),
467 				oReader = new filesystem.DirectoryReader(this._entry.createReader()),
468 				
469 				aDirContent = [],
470 			
471 				// define recursive function to read all content
472 				fnReadBlock = function(oD, oReader, aContent) {
473 				
474 					oReader.readEntries()
475 						.done($.proxy(
476 							function(oDeferred, oReader, aList, aResults) {
477 							
478 								var i = 0, len = aResults.length;
479 								
480 								for ( ; i < len ; ++i) aList.push(aResults[i]);
481 								
482 								if (len) {
483 									// fetch next block
484 									fnReadBlock(oDeferred, oReader, aList);
485 								} else {
486 									// end of content reading
487 									oDeferred.resolve(aList);
488 								}
489 							
490 							}, this, oD, oReader, aContent)
491 						)
492 						.fail(filesystem._fnErrorCallback(oD));
493 				};
494 			
495 			
496 			// start reading content
497 			fnReadBlock(oDef, oReader, aDirContent);
498 			
499 			return oDef;
500 		};
